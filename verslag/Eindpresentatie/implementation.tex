    \begin{minipage}{0.4\textwidth}
        \begin{itemize}
            \only<1-> {\item Invert $A^i$ separately }
                  \only<1-2>{
                      \begin{itemize}
                          \item Fast
                          \item Numerically unstable
                      \end{itemize}
                  }
                  \only<3-> { \item Full inversion }
                  \only<3>{
                      \begin{itemize}
                          \item Slow
                          \item Stable for pseudoinverse
                      \end{itemize}
                  }
                  \only<4-> { \item Sparse full inversion }
                  \only<4>{
                      \begin{itemize}
                          \item $A^i = U^i \Sigma^i V^{i\dagger}$
                      \end{itemize}
                  }
        \end{itemize}
